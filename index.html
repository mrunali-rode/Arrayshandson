Que 9. Unique Color Shirt

Ans: function Unique_Shirts (arr,N) {

                let count=0;
               for( i=0;i < N;i++){
                 for( j=0;j< N ;j++){
                   if((arr[i]==arr[j]) && i != j)
                    break;
                 }
                 if(j==N){
                  count++;
                 }
               }
               return count;
              }
